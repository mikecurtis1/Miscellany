ID's = [ [ comma-delimited profile array keys ] ; [ [ TRUE | FALSE ] include/exclude the matches in results ] ]

FILTER = [ [ pipe-delimited profile array keys ] ; [ string to match | flags [ TRUE | FALSE | NULL | ISSET] ] ; [ [ TRUE | FALSE ] include/exclude the matches in results ] ]

	Flags: 
		- TRUE, FALSE, and NULL submitted as text strings, but will be cast to proper datatypes for matching
		- ISSET checks if the key is an array set in the profile

VALUE = [ pipe-delimited multi-dimensional array keys ]

Examples: 

	#1 IDS Lenders with z3950 availability
	
		FILTER 1 = [ availability|type ; z3950 ; TRUE ]
		FILTER 2 = [ ids_lender ; FALSE ; FALSE ]

	#2 IDS Lenders without an OCLC wskey
		
		FILTER 1 = [ wskey ; NULL ; TRUE]
		FILTER 2 = [ ids_lender ; FALSE ; FALSE ]

	#3 Profiles with custom links array set
	
		FILTER 1 = [ custom_links ; ISSET ; TRUE ]
		
	#4 Values can include a combination of string or numeric keys
	
		VALUE 1 = [ wskey ]
		VALUE 2 = [ availability|type ]
		VALUE 3 = [ custom_links|links|0|url ]
